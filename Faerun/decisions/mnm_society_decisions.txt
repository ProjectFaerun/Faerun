	assassins_borrow_money = {
		only_playable = yes
		is_in_society = yes
		ai_check_interval = 80
		potential = {
			not_casting = yes
			has_dlc = "Mystics"
				society_member_of = eldreth_veluuthra

				society_rank = {
					society = eldreth_veluuthra
					rank == 4
				}
			}
			NOT = { has_character_modifier = assassins_debt }
		}

		allow = {
			custom_tooltip = {
				text = assassins_borrow_money_tooltip

				any_society_member = {
					is_society_grandmaster = yes
					society_member_of = eldreth_veluuthra
					OR = {
						ai = no
						NOT = { primary_title = { title = d_victorious_blade } }
						AND = {
							primary_title = { title = d_victorious_blade }
							wealth = 50
						}
					}
				}
			}
		}

            increase_visibility_effect = yes

						is_society_grandmaster = yes
						society_member_of = eldreth_veluuthra
					}
					character_event = { id = MNM.6000 }
				}
			}
		}

		ai_will_do = {
			factor = 1

			modifier = {
				factor = 0.5 #Slow down
			}
			modifier = {
				factor = 0
				wealth = 1
			}
			modifier = {
				factor = 0.1
				war = no
			}

	}

				factor = 0.1
				war = yes
			}

				factor = 0
				has_character_flag = assassins_refused_to_pay_debt
			}
			modifier = {
				factor = 0

					AND = {
						is_variable_equal = { which = assassins_debt value = 50 }

							NOT = { wealth = 50 }

								NOT = { monthly_income = 0 }

						is_variable_equal = { which = assassins_debt value = 250 }

								NOT = { wealth = 1000 }

						}
