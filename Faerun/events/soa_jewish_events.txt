# Jews Expelled
character_event = {
	id = SoA.105
	desc = EVTDESC_SoA_105_GUR
	picture = GFX_evt_jewish_market
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	immediate = {
		any_realm_lord = {
			limit = {
				OR = {
					culture = gur
					culture = gurri
				}
				ai = yes
			}
			any_demesne_title = {
				grant_title_no_opinion = ROOT
			}
		}
	}

	option = {
		name = EVTOPTA_SoA_105_GUR
		hidden_tooltip = {
			any_realm_lord = {
				limit = { 
					OR = {
						ai = no
						has_character_modifier = borrowed_from_gurs
					}
				}
				character_event = { id = SoA.107 }
			}
		}
	}
}

### Jewish Court Events

# Jewish Diplomat Appears at Court (Ashkenazi)
character_event = {
	id = SoA.200
	desc = EVTDESC_SoA_200_GUR
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	prisoner = no
	only_playable = yes
	capable_only = yes
	only_independent = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		OR = {
			AND = {
				culture_group = abeiran_group
				NOT = {
					location = {
						region = laerakond_region
					}
				}
			}
			culture_group = calishite_group
			culture_group = chondathan_group
			culture_group = damaran_group
			culture_group = illuskan_group
			culture_group = imaskari_group
			culture = chultan
			culture = thinguth
			culture = lantanna
			culture_group = lapal_group
			culture_group = mulan_group
			culture_group = netherese_group
			culture_group = raum_group
			culture_group = tethyrian_group
		}
		NOR = { 
			culture = gur 
			culture = gurri
		}
		NOT = { government = order_government }
		NOT = { government = nomadic_tribal_government }
		NOT = { government = tribal_government }
		NOT = { has_character_modifier = expelled_gurs }
		NOT = { trait = incapable }
		NOT = { is_inaccessible_trigger = yes }

		trigger_if = {
			limit = { ai = yes }

			free_court_slots >= 5
		}
		is_landed = yes
	}
	
	mean_time_to_happen = {
		months = 1100
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			female = random
			culture = gurri
			dynasty = culture
			religion = joyful
			trait = patron_selune
			#female = no
			age = 30
			trait = charismatic_negotiator
			attributes = {
				diplomacy = 10
			}
			flag = ai_flag_refuse_conversion
			flag = special_courtier
		}
		new_character = {
			remove_trait = socializer
			remove_trait = gamer
			remove_trait = hedonist
			remove_trait = duelist
			remove_trait = hunter
			remove_trait = strategist
			remove_trait = administrator
			remove_trait = architect
			remove_trait = gardener
			remove_trait = schemer
			remove_trait = impaler
			remove_trait = seductress
			remove_trait = scholar
			remove_trait = theologian
			remove_trait = mystic
			random_list = {
				70 = {
					add_trait = socializer
				}
				30 = {
					add_trait = socializer
					add_trait = shrewd
				}
			}
			save_event_target_as = jewish_courtier
		}
	}
	
	option = {
		name = EVTOPTA_SoA_200
		custom_tooltip = {
			text = gur_courtier_tooltip
		}
	}
}

# Chancellor (Hidden)
character_event = {
	id = SoA.560
	hide_window = yes

	min_age = 16
	only_capable = yes
	prisoner = no
	
	trigger = {
		has_job_title = job_chancellor
		OR = {
			culture = gur 
			culture = gurri
		}
		employer = {
			NOR = {
				culture = gur 
				culture = gurri
			}
		}
	}
	
	mean_time_to_happen = {
		months = 280
		modifier = {
			factor = 0.5
			OR = {
				has_focus = focus_scholarship
				liege = { has_focus = focus_scholarship }
			}
		}
	}
	
	immediate = {
		employer = {
			character_event = { id = SoA.561 }
		}
	}
}
