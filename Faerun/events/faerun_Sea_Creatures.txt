character_event = {
	#Sahuagin aging
	id = FSC.0003
	hide_window = yes
	is_triggered_only = yes

	only_men = yes
	min_age = 30
	culture = sahuagin

	trigger = {
		trait = creature_sahuagin
	}

	immediate = {
		if = {
			limit = { age >= 1000 }
			if = {
				limit = { NOT = { trait = sahuagin_emperor } }
				add_trait = sahuagin_emperor
				remove_trait = sahuagin_king
			}
		}
		else_if = {
			limit = { age >= 600 }
			if = {
				limit = { NOT = { trait = sahuagin_king } }
				add_trait = sahuagin_king
				remove_trait = sahuagin_prince
			}
		}
		else_if = {
			limit = { age >= 280 }
			if = {
				limit = { NOT = { trait = sahuagin_prince } }
				add_trait = sahuagin_prince
				remove_trait = sahuagin_baron
			}
		}
		else_if = {
			limit = { age >= 150 }
			if = {
				limit = { NOT = { trait = sahuagin_baron } }
				add_trait = sahuagin_baron
				remove_trait = sahuagin_chief
			}
		}
		else_if = {
			limit = { age >= 70 }
			if = {
				limit = { NOT = { trait = sahuagin_chief } }
				add_trait = sahuagin_chief
				remove_trait = sahuagin_warrior
			}
		}
		else_if = {
			limit = { NOT = { trait = sahuagin_warrior } }
			add_trait = sahuagin_warrior
		}
	}
}
