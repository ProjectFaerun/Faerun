#From on_character_ask_to_join_society
character_event = {
    id = LA.100
    hide_window = yes
    is_triggered_only = yes
    trigger = {
    	FROM = { leader = { society_member_of = lord_alliance } }
    }
    immediate = {
    	if = { #Find a non-prisoner member
    		limit = { FROM = { leader = { prisoner = yes } } }
    		FROM = {
    			any_society_member = {
    				limit = {
    					society_member_of = lord_alliance
    					society_rank == 4
    					prisoner = no
    				}
    				character_event = { id = LA.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = lord_alliance
    					society_rank == 3
    					prisoner = no
    				}
    				character_event = { id = LA.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = lord_alliance
    					society_rank == 2
    					prisoner = no
    				}
    				character_event = { id = LA.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = lord_alliance
    					society_rank == 1
    					prisoner = no
    				}
    				character_event = { id = LA.101 }
    				break = yes
    			}
    		}
			#Everyone is somehow in prison so just use teh leader anyway
    	}
    	FROM = { leader = { character_event = { id = LA.101 } } }
    }
}
