character_event = {
	#Dragon Aging
	id = dragon_type.1
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		is_dragon = yes
	}

	option = {
		if = {
			limit = {
				age = 800
			}
			add_trait = dragon_ancient
			remove_trait = dragon_adult
		}
		else_if = {
			limit = {
				age = 100
			}
			add_trait = dragon_adult
			remove_trait = dragon_young
		}
		else_if = {
			limit = {
				age = 6
			}
			add_trait = dragon_young
			remove_trait = dragon_wyrmling
		}
		else_if = {
			limit = {
				NOR = {
					trait = dragon_wyrmling
					trait = dragon_young
					trait = dragon_adult
					trait = dragon_ancient
				}
				age = 0
			}
			add_trait = dragon_wyrmling
		}
		else = {

		}
	}
}
