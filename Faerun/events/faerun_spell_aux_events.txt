character_event = { #dragonbreath for AI
	id = spellaux.42
	picture = GFX_evt_blood_war_1
	desc = SPELLAUX.42
	is_triggered_only = yes

	only_capable = yes
	min_age = 16
	ai = yes
	war = yes


	trigger = {
		OR = {
			AND = { #breath weapon
				trait = creature_dragon
				OR = {
					NOT = { has_character_flag = breath_weapon_cooldown }
					had_character_flag = { flag = breath_weapon_cooldown months = 1 }
				}
			}
		}
		NOR = {
			trait = wounded
			trait = severely_injured
			trait = infection
			trait = mangled
			trait = maimed
			trait = infirm

			trait = one_legged

			trait = flu
			trait = dysentery
			trait = gout
			trait = food_poisoning
			trait = cancer
			trait = scurvy
		}
	}

	option = { #breath weapon
		name = spellaux.42.breath.weapon
		trigger = {
			AND = {
				trait = creature_dragon
				age = 16
				OR = {
					NOT = { has_character_flag = breath_weapon_cooldown }
					had_character_flag = { flag = breath_weapon_cooldown months = 1 }
				}
			}
		}
		set_character_flag = breath_weapon_cooldown
		hidden_tooltip = {
			combat = {
				enemy = {
					any_unit_leader = {
						repeat_event = { id = spellaux.24 }
					}
				}
			}
		}
		any_current_enemy = {
			random_realm_character = {
				limit = {
					at_location = ROOT
					in_command = yes
					prisoner = no
				}
				if = {
					limit = {
						ROOT = {
							age = 16
							NOT = { age = 26 }
						}
					}
					troops = -0.02
				}

				else_if = {
					limit = {
						ROOT = {
							age = 26
							NOT = { age = 51 }
						}
					}
					troops = -0.04
				}

				else_if = {
					limit = {
						ROOT = {
							age = 51
							NOT = { age = 101 }
						}
					}
					troops = -0.06
				}

				else_if = {
					limit = {

					troops = -0.08
				}

				else_if = {
					limit = {
						ROOT = {
							age = 201
							NOT = { age = 401 }
						}
					}
					troops = -0.1
				}

				else_if = {
					limit = {
						ROOT = {
							age = 401

						}

				}

		if = {
			limit = {

		else = {

character_event = { #Activates or kills clones on creator's death
	id = spellaux.44
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		any_child = {
			trait = is_clone
			trait = stasis_clone
		}
	}

	immediate = {
		any_child = {
			if = {
				limit = {
					trait = is_clone
					trait = stasis_clone
				}
				remove_trait = stasis_clone
				random_list = {
					90 = {
						#nothing
					}
					10 = {
						death = {
							death_reason = death_missing
						}
					}
				}
			}
		}
		recalc_succession = yes
		player_heir = {
			if = {
				limit = {
					trait = is_clone
				}
				prompt_name = {
					player = ROOT
					type = RENAME_CLONE_MESSAGE
				}
			}
		}
	}
}
