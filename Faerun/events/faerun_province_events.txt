#Azure Druid
province_event = {
	id = frprov.4
	desc = FRPROV.4.D
	picture = GFX_evt_heretic
	
	is_triggered_only = yes
	
	immediate = {
		add_province_modifier = {
			name = recent_revolt
			years = 1
		}
		create_character = {
			age = 40
			attributes = {
				learning = 6
			}
			add_trait = journeyman_druid
			add_trait = spellscar
			add_trait = zealous
			random_traits = yes
			female = 50
			dynasty = actually_culture
		}
		new_character = {
			religion = blue_flame
			culture = ROOT
		}
		holder_scope = {
			new_character = {
				add_spell_scar_effect = yes
				spawn_unit = {
					province = ROOT
					home = ROOT
					owner = THIS
					leader = THIS
					troops = {
						light_infantry = { 40 40 }
						archers = { 15 15 }
						light_cavalry = { 5 5 }
						heavy_infantry = { 5 5 }
						mob_troops = { 100 100 }
					}
					scaled_by_biggest_garrison = 1.5 # Balance this!
					attrition = 0
					maintenance_multiplier = 0
					reinforces = no
					disband_on_peace = yes
					cannot_inherit = yes
					can_toggle_looting = no
				}
				war = {
					target = PREV
					casus_belli = heretic_revolt
					thirdparty_title = ROOT
				}
			}
		}
	}
	
	option = {
		name = FRPROV.4.O
	}
}
