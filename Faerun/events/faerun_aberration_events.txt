character_event = { # Beholder pregnancy starts
	id = aberrationevt.3
	desc = BEHOLDERPREG.1
	picture = GFX_evt_underdark_1
	
	trigger = {
		NOT = { trait = tb_sterile }
		is_incapable = no
		prisoner = no
		age = 30
		trait = creature_beholder
		OR = {
			is_landed = yes
			any_close_relative = {
				is_landed = yes
			}
		}
	}
	
	is_triggered_only = yes
	
	option = {
		name = BEHOLDERPREG.1a
		random = {
			chance = 80
			add_trait = paranoid
		}
		hidden_effect = {
			character_event = { id = aberrationevt.4 months = 6 }
			add_trait = tb_sterile
		}
	}
}

character_event = { # Beholder pregnancy ends
	id = aberrationevt.4
	desc = BEHOLDERPREG.2
	picture = GFX_evt_underdark_1
	
	is_triggered_only = yes
	
	option = {
		name = BEHOLDERPREG.2a
		random = {
			chance = 30
			remove_trait = paranoid
		}
		if = {
			limit = {
				is_female = no
			}
			create_character = {
				age = 0
				female = random
				culture = beholder
				religion = ROOT
				dynasty = ROOT
				trait = creature_beholder
				genetic_father = ROOT
				recalc_succession = yes
			}
			new_character = {
				set_father = ROOT
				culture = ROOT
			}
		}
		else = {
			create_character = {
				age = 0
				female = random
				culture = beholder
				religion = ROOT
				dynasty = ROOT
				trait = creature_beholder
				genetic_mother = ROOT
				recalc_succession = yes
			}
			new_character = {
				set_mother = ROOT
				culture = ROOT
			}
		}
		gfx_culture_scope = {
			new_character = {
				set_graphical_culture = PREV
				random = {
					chance = 5
					add_trait = elder_orb
					add_trait = giant
				}
			}
		}
	}
}
