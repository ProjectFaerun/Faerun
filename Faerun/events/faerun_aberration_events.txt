character_event = { # Beholder pregnancy starts
	id = aberrationevt.3
	desc = BEHOLDERPREG.1
	picture = GFX_evt_underdark_1
	is_triggered_only = yes

	only_capable = yes
	prisoner = no
	min_age = 30
	
	trigger = {
		trait = creature_beholder
		NOT = { trait = tb_sterile }
		OR = {
			is_landed = yes
			any_close_relative = {
				is_landed = yes
			}
		}
		trigger_if = {
			limit = { ai = yes }

			NOT = {
				any_child = {
					count >= 3
				}
			}
		}
	}

	option = {
		name = BEHOLDERPREG.1a
		random = {
			chance = 80
			add_trait = paranoid
		}
		hidden_effect = {
			character_event = { id = aberrationevt.4 months = 6 }
			add_trait = tb_sterile
		}
	}
}

character_event = { # Beholder pregnancy ends
	id = aberrationevt.4
	desc = BEHOLDERPREG.2
	picture = GFX_evt_underdark_1
	
	is_triggered_only = yes
	
	option = {
		name = BEHOLDERPREG.2a
		random = {
			chance = 30
			remove_trait = paranoid
		}

		gfx_culture_scope = {
			culture_scope = { save_event_target_as = source_culture }
		}

		if = {
			limit = {
				is_female = no
			}
			create_character = {
				age = 0
				female = random
				culture = beholder
				race = event_target:source_culture
				dynasty = ROOT
				trait = creature_beholder
				genetic_father = ROOT
				recalc_succession = yes
			}
			new_character = {
				set_father = ROOT
			}
		}
		else = {
			create_character = {
				age = 0
				female = random
				culture = beholder
				race = event_target:source_culture
				dynasty = ROOT
				trait = creature_beholder
				genetic_mother = ROOT
				recalc_succession = yes
			}
			new_character = {
				set_mother = ROOT
			}
		}
		new_character = {
			culture = ROOT
			religion = ROOT
			random = {
				chance = 5
				add_trait = elder_orb
				add_trait = giant
			}
		}
	}
}
