namespace = faithcleanup

# Lose some titular titles if holder is inadequate
character_event = {
	id = faithcleanup.0
	hide_window = yes
	is_triggered_only = yes

	only_playable = yes

	trigger = {
		NOT = { has_character_flag = reverting_religion_change }
		
		trigger_if = {
			limit = { has_landed_title = k_army_of_shadows }

			OR = {
				is_evil_religion = no
				is_undead = no
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = k_hordes_of_dragonspear }

			NOR = {
				religion = infernal
				culture = baatezu
				culture = abishai
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = e_a_brotherhood }

			olrox_artifact_purearcane_requirement = no
		}
		trigger_else_if = {
			limit = { has_landed_title = e_brotherhood_of_true_flame }

			NOR = {
				olrox_artifact_purearcane_requirement = yes
				z_shadow = yes
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_shar }

			OR = {
				NOR = {
					religion = shadow_gods
					religion = calishite_faith
				}
				NOR = {
					trait = patron_shar
					has_patron_god = no
				}
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_cult_of_shajar }

			OR = {
				NOR = {
					religion = nogari
					religion = nogari_reformed
				}
				NOR = {
					trait = patron_shajar
					has_patron_god = no
				}
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_cult_of_ragarra }

			OR = {
				NOR = {
					religion = nogari
					religion = nogari_reformed
				}
				NOR = {
					trait = patron_ragarra
					has_patron_god = no
				}
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_cult_of_tiamat }

			NOR = {
				AND = {
					OR = {
						religion = draconic_pantheon
						religion = lapal_faith
					}
					OR = {
						trait = patron_tiamat
						has_patron_god = no
					}
				}
				religion = tiamat
				religion = chessic
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_cult_of_gilgeam }

			NOT = { religion = untheric_faith }
		}
		trigger_else_if = {
			limit = { has_landed_title = d_priesthood_of_ishtar }

			OR = {
				NOR = {
					religion = untheric_faith
					religion = mulhorandi_faith
				}
				NOR = {
					trait = patron_isis
					has_patron_god = no
				}
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = d_fallen_justice }

			OR = {
				NOR = {
					religion = untheric_faith
					religion = mulhorandi_faith
				}
				NOR = {
					trait = patron_anhur
					has_patron_god = no
				}
			}
		}
		trigger_else_if = {
			limit = { has_landed_title = k_grand_caliphate }

			NOR = {
				religion = ten_thousand_gods
				religion = qisma
				religion = pantheist_league
				religion = muhjari
				religion = kahin
			}
		}
		trigger_else = {
			always = no
		}
	}

	immediate = {
		if = {
			limit = {
				has_landed_title = k_army_of_shadows
				OR = {
					is_evil_religion = no
					is_undead = no
				}
			}
			character_event = { id = frstory.16 }
		}
		else_if = {
			limit = {
				has_landed_title = k_hordes_of_dragonspear
				NOR = {
					religion = infernal
					culture = baatezu
					culture = abishai
					trait = warlock_fiend
				}
			}
			character_event = { id = frstory.77 }
		}
		else_if = {
			limit = {
				has_landed_title = e_a_brotherhood
				olrox_artifact_purearcane_requirement = no
			}
			character_event = { id = BRO.6000 }
		}
		else_if = {
			limit = {
				has_landed_title = e_brotherhood_of_true_flame
				NOR = {
					olrox_artifact_purearcane_requirement = yes
					z_shadow = yes
				}
			}
			character_event = { id = BRO.6001 }
		}
		else_if = {
			limit = {
				has_landed_title = d_shar
				OR = {
					NOR = {
						religion = shadow_gods
						religion = calishite_faith
					}
					NOR = {
						trait = patron_shar
						has_patron_god = no
					}
				}
			}
			character_event = { id = faithcleanup.8 }
		}	
		else_if = {
			limit = {
				has_landed_title = d_cult_of_shajar
				NAND = {
					OR = {
						religion = nogari
						religion = nogari_reformed
					}
					OR = {
						trait = patron_shajar
						has_patron_god = no
					}
				}
			}
			character_event = { id = faithcleanup.1 }
		}
		else_if = {
			limit = {
				has_landed_title = d_cult_of_tiamat
				NOR = {
					AND = {
						OR = {
							religion = draconic_pantheon
							religion = lapal_faith
						}
						OR = {
							trait = patron_tiamat
							has_patron_god = no
						}
					}
					religion = tiamat
					religion = chessic
				}
			}
			character_event = { id = faithcleanup.2 }
		}
		else_if = {
			limit = {
				has_landed_title = d_cult_of_gilgeam
				NOT = { religion = untheric_faith }
			}
			character_event = { id = faithcleanup.3 }
		}
		else_if = {
			limit = {
				has_landed_title = d_priesthood_of_ishtar
				OR = {
					NOR = {
						religion = untheric_faith
						religion = mulhorandi_faith
					}
					NOR = {
						trait = patron_anhur
						has_patron_god = no
					}
				}
			}
			character_event = { id = faithcleanup.4 }
		}
		else_if = {
			limit = {
				has_landed_title = d_fallen_justice
				OR = {
					NOR = {
						religion = untheric_faith
						religion = mulhorandi_faith
					}
					NOR = {
						trait = patron_anhur
						has_patron_god = no
					}
				}
			}
			character_event = { id = faithcleanup.5 }
		}
		else_if = {
			limit = {
				has_landed_title = d_cult_of_ragarra
				NAND = {
					OR = {
						religion = nogari
						religion = nogari_reformed
					}
					OR = {
						trait = patron_ragarra
						has_patron_god = no
					}
				}
			}
			character_event = { id = faithcleanup.6 }
		}
		else_if = {
			limit = {
				has_landed_title = k_grand_caliphate
				NOR = {
					religion = ten_thousand_gods
					religion = qisma
					religion = pantheist_league
					religion = muhjari
					religion = kahin
				}
			}
			character_event = { id = faithcleanup.7 }
		}
	}
}
