namespace = courtflag

character_event = {
	id = courtflag.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		trigger_if = {
			limit = { FROMFROM = { has_character_flag = vampire_court } }

			OR = {
				trait = vampire
				trait = vampire_spawn
			}
		}
		trigger_else_if = {
			limit = { FROMFROM = { has_character_flag = undead_court } }

			OR = {
				trait = undead
				trait = lich
				trait = vampire
				trait = vampire_spawn
			}
		}
		trigger_else_if = {
			limit = { FROMFROM = { has_character_flag = lycan_court } }

			is_werebeast = yes
		}
		trigger_else = {
			always = no
		}
	}
	
	immediate = {
		if = {
			limit = {
				FROMFROM = {
					has_character_flag = vampire_court
				}
				OR = {
					trait = vampire
					trait = vampire_spawn
				}
			}
			set_character_flag = vampire_court
		}
		if = {
			limit = {
				FROMFROM = {
					has_character_flag = undead_court
				}
				OR = {
					trait = undead
					trait = lich
					trait = vampire
					trait = vampire_spawn
				} 
			}
			set_character_flag = undead_court
		}
		if = {
			limit = {
				FROMFROM = {
					has_character_flag = lycan_court
				}
				is_werebeast = yes
			}
			set_character_flag = lycan_court
		}
	}
}
