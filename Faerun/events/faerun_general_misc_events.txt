# New Lotus Emperor
character_event = {
	id = generalmisc.51
	desc = GENMISC.51
	picture = GFX_evt_yikarian

	is_triggered_only = yes

	immediate = {
		change_variable = {
			which = global_counter_divine
			value = 1
		}
	}
	option = {
		name = GENMISC.51a
		piety = 400
		create_character = {
			age = 40
			random_traits = yes
			trait = trained_cleric
			trait = zealous
			trait = creature_yikarian
			religion = faceless_god
			culture = yikarian
			dynasty = culture
		}
		new_character = {
			grant_title = k_lotus_throne
			k_lotus_throne = {
				add_law = agnatic_succession
				add_law = succ_yikaria
			}
			if = {
				limit = {
					ROOT = { tier = EMPEROR }
				}
				set_defacto_liege = ROOT
			}
		}
	}
}

	id = generalmisc.53
	picture = GFX_evt_busy_trading_dock_republic
	is_triggered_only = yes
	desc = {
		text = EVTDESC_GENMISC.53
	}
	option = {
		name = GENMISC.53a
		custom_tooltip = {
			text = GENMISC.53a_TOOLTIP
			any_vassal = {
				if = {
					limit = {
						government = democratic_feudal_government
					}
					opinion = {
						modifier = opinion_sponsored_merchant_republic_bad
						who = ROOT
					}
				}
				else = {
					opinion = {
						modifier = opinion_sponsored_merchant_republic_good
						who = ROOT
					}
				}
			}
		}
		
	}
}
