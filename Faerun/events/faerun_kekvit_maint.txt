# Monster Mobs for all monster provinces
character_event = {
	id = KekvitMaint.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_save_game = no # First startup of the game
	}
	immediate = {
		while = {
			limit = {
				any_title = {
					tier = COUNT
					location = {
						culture = monster
						holder = 0
					}
				}
			}
			any_title = {
				limit = {
					tier = COUNT
					location = {
						culture = monster
						holder = 0
					}
				}
				create_character = {
					name = "Monster Mob"
					dynasty = none
					culture = monster
					race = monster
				}
				new_character = {
					grant_title = PREV
				}
			}
		}
	}
}
			
# Sets startup flag, for submod compatibility
character_event = {
	id = KekvitMaint.2
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_save_game = no
		NOT = { has_global_flag = faerun_started }
	}
	immediate = {
		set_global_flag = faerun_started
	}
}
