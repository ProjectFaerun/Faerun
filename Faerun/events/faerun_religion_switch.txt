character_event = {
	id = rlg_chg.10
	hide_window = yes

	is_triggered_only = yes

	only_playable = yes
	#only_independent = yes #Unsure

	trigger = {
		higher_real_tier_than = COUNT
		NOR = {
			holy_order = yes
			culture = monster
			religion = monster_pantheon
		}
		has_game_rule = {
			name = diversity
			value = on
		}
	}

	immediate = {
		export_to_variable = {
			which = local_number_of_prov
			value = num_of_count_titles_in_realm
		}
		divide_variable = {
		    which = local_number_of_prov
		    value = 1.65
		}
		if = { #culture diversity activate
			limit = {
				AND = {
					num_culture_realm_provs = {
					    culture = ROOT
					    value <= local_number_of_prov
					}
					any_vassal = {
						is_powerful_vassal = yes
						NOT = { culture = ROOT }
					}
					any_realm_province = { culture = ROOT }
				}
				NAND = {
					num_religion_realm_provs = {
							religion = ROOT
							value <= local_number_of_prov
					}
					any_vassal = {
						is_powerful_vassal = yes
						NOT = { religion = ROOT }
					}
					any_realm_province = { religion = ROOT }
				}
				NOR = {
					has_character_modifier = cult_diversity_mod
					has_character_modifier = total_diversity_mod
					has_character_modifier = undead_realm_lord
				}
			}
			character_event = { id = rlg_chg.11 }
		}
		else_if = { #religion diversity activate
			limit = {
				NAND = {
					num_culture_realm_provs = {

						NOT = { religion = ROOT }
					}
					any_realm_province = { religion = ROOT }
				}
				NOR = {
					has_character_modifier = rel_diversity_mod
					has_character_modifier = total_diversity_mod
					has_law = revoke_title_law_2
					has_character_modifier = undead_realm_lord

				}

					has_character_modifier = total_diversity_mod

					has_character_modifier = undead_realm_lord
				}
			}
			character_event = { id = rlg_chg.13 }
		}
		if = { #strip culture diversity
			limit = {
				has_character_modifier = cult_diversity_mod
				OR = {
					has_character_modifier = undead_realm_lord

					has_character_modifier = undead_realm_lord

						num_religion_realm_provs = {
