namespace = frimmortal

# Vampire hunter
character_event = {
	id = frimmortal.1
	desc = FRIMMORTAL.1.D
	picture = GFX_evt_bad_news
	border = GFX_event_normal_frame_intrigue
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_character_modifier = recent_duel_timer }
		OR = {
			trait = vampire
			trait = vampire_spawn
		}
		OR = {
			NOT = { has_character_flag = vampire_hunter_spawned }
			had_character_flag = {
				flag = vampire_hunter_spawned
				years > 5
			}
		}
		location = { has_owner = yes } # not commanding troops in sea
	}
	
	weight_multiplier = {
		factor = 5
		mult_modifier = {
			factor = 0.5
			trait = vampire_spawn
		}
		mult_modifier = {
			factor = 2
			is_ruler = no
		}
		mult_modifier = {
			factor = 1.5
			has_character_modifier = known_vamp_modifier
		}
		mult_modifier = {
			factor = 0.5
			intrigue >= 10
		}
		mult_modifier = {
			factor = 0.5 # stacks with the above
			intrigue >= 15
		}
		mult_modifier = {
			factor = 2
			intrigue < 6
		}
		mult_modifier = {
			factor = 4
			intrigue < 1
		}
	}
	
	immediate = {
		clr_character_flag = vampire_hunter_spawned
		set_character_flag = vampire_hunter_spawned
		location = { save_event_target_as = vamp_location }
	}
	
	option = {
		trigger = {
			is_ruler = yes
			wealth = 50
		}
		name = FRIMMORTAL.1.Oa
		wealth = -50
		hidden_tooltip = {
			random_list = {
				1 = {
					mult_modifier = { # these should all stack
						factor = 1.5
						intrigue >= 7
					}
					mult_modifier = {
						factor = 1.5
						intrigue >= 10
					}
					mult_modifier = {
						factor = 1.5
						intrigue >= 15
					}
					mult_modifier = {
						factor = 1.5
						intrigue >= 20
					}
					mult_modifier = {
						factor = 1.5
						intrigue >= 25
					}
					hidden_tooltip = {
						character_event = { id = frimmortal.2 days = 10 }
					}
				}
				5 = {
					set_character_flag = failed_killing_vampire_hunter
					hidden_tooltip = {
						character_event = { id = frimmortal.3 days = 10 }
					}
				}
			}
		}
	}
	option = {
		name = FRIMMORTAL.1.Ob
		hidden_tooltip = {
			character_event = { id = frimmortal.3 days = 10 }
		}
	}
}
