character_event = {
	id = seldarine.5
	picture = GFX_evt_savage_frontier
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_courtier = {
			has_character_flag = reverend_one
			is_alive = yes
		}
		OR = {
			war = no
			OR = {
				is_primary_war_defender = yes
				AND = {
					is_primary_war_attacker = yes
					any_current_enemy = {
						is_primary_war_defender = yes
						OR = {
							trait = patron_malar
							trait = patron_ghaunadaur
							religion = ghaunadaur
							religion = vhaeraun
							religion = kiaransalee
							trait = patron_talos
							trait = patron_auppenser
							religion = lolth_cult
							orc_religion_trigger = yes
							goblin_religion_trigger = yes
							religion = vaprak
							religion = infernal
							religion = abyssal
							religion = entropy
							religion = faithless
							trait = patron_hyrsam
						}
					}
				}
			}
		}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				has_character_flag = reverend_one
			}
			death = { death_reason = death_planes }
			if = {
				limit = {
					liege = {
						ai = no
					}
				}
				liege = {
					character_event = { id = seldarine.6 }
				}
			}
		}
	}
}
