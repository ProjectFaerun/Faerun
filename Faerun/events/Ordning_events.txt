namespace = Ordning

###########################################
# Arrival of guru                         #
###########################################

# Start looking for guru
character_event = {
	id = Ordning.0001
	desc = EVTDESC_Ordning_0001
	picture = GFX_evt_runecaster_1
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTA_Ordning_0001
		hidden_tooltip = { character_event = { id = Ordning.0002 days = 30 random = 60 } }
	}
}

# Guru is found
character_event = {
	id = Ordning.0002

	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		clr_character_flag = looking_for_guru
		set_character_flag = found_guru
		random_list = {
			15 = { # Finds great guru
				create_character = {
					random_traits = no
					dynasty = NONE
					religion = ROOT
					female = 50
					age = 100
					attributes = {
						diplomacy = 10
						stewardship = 10
						intrigue = 10
						learning = 18
					}
					trait = genius
					trait = scholar
					trait = patient
					trait = mastermind_theologian
					trait = creature_giant
				}
				new_character = {
					set_character_flag = guru
					give_minor_title = title_runecaster
					piety = 600
					prestige = 150
					if = {
						random_list = {
							60 = { add_trait = master_wizard }
							25 = { add_trait = master_sorcerer }
							15 = { add_trait = master_bard } 
						}
					}
					if = {
						random_list = {
							60 = { culture = storm_giant }
							20 = { culture = cloud_giant }
							10 = { culture = stone_giant }
							5 = { culture = fire_giant }
							3 = { culture = frost_giant }
							1 = { culture = hill_giant }
							1 = { culture = ogre }
						}
					}
					opinion = {
						modifier = opinion_loyal_servant
						who = ROOT
						years = 150
					}
					random_list = {  # Small chance of having some unusual memorable trait
						1 = { add_trait = blinded }
						1 = { add_trait = dwarf }
						50 = {}
					}
					character_event = { id = Ordning.0003 }
				}
			}
			80 = { # Finds ordinary guru
				create_character = {
					random_traits = no
					dynasty = NONE
					religion = ROOT
					female = 50
					age = 75
					attributes = {
						diplomacy = 10
						stewardship = 8
						intrigue = 9
						learning = 15
					}
					trait = quick
					trait = scholar
					trait = patient
					trait = mastermind_theologian
					trait = creature_giant
				}
				new_character = {
					set_character_flag = guru
					give_minor_title = title_guru
					piety = 300
					prestige = 750
					if = {
						random_list = {
							60 = { add_trait = expert_wizard }
							25 = { add_trait = expert_sorcerer }
							15 = { add_trait = expert_bard } 
						}
					}
					if = {
						random_list = {
							85 = { culture = storm_giant }
							10 = { culture = cloud_giant }
							5 = { culture = stone_giant }
						}
					}
					opinion = {
						modifier = opinion_loyal_servant
						who = ROOT
						years = 100
					}
					character_event = { id = Ordning.0003 }
				}
			}
			5 = { # Finds con artist pretending to be guru - but doesn't know it yet...
				create_character = {
					random_traits = no
					dynasty = NONE
					religion = ROOT
					culture = hill_giant
					female = 50
					age = 40
					attributes = {
						diplomacy = 12
						stewardship = 8
						intrigue = 16
						learning = 10
					}
					trait = quick
					trait = martial_cleric
					trait = weak
					trait = deceitful
				}
				new_character = {
					set_character_flag = guru
					set_character_flag = fake_guru
					give_minor_title = title_runecaster
					piety = 100
					prestige = 50
					add_trait = creature_giant
					opinion = {
						modifier = opinion_loyal_servant
						who = ROOT
						years = 50
					}
					character_event = { id = Ordning.0003 }
				}
			}
		}
	}
}

# Guru bounce event
character_event = {
	id = Ordning.0003

	hide_window = yes
	
	is_triggered_only = yes

	immediate = {
		FROM = { character_event = { id = Ordning.0004 } }
	}
}
