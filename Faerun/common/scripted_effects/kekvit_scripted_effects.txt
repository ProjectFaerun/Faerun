			limit = {

				1 = {
					add_trait = afflicted_lycanthrope
				}

					add_trait = natural_lycanthrope

			}
			if = {
				limit = {

				}
				add_trait = werewolf
			}
			else_if = {
				limit = {
					PREV = { trait = wererat }

				add_trait = wererat
			}
			else_if = {
				limit = {

			}

			}
			update_hidden_modifiers = yes

	}

rd_age_up_dwarf = {
	if = {
		limit = {
			age = 40
		}
		break = yes
	}
	else = {
		set_variable = {
			which = local_final_age
			value = 40
		}
		export_to_variable = {
			which = local_starting_age
			value = age
		}
		subtract_variable = {
			which = local_final_age
			which = local_starting_age
		}
		add_age = local_final_age
	}
}
