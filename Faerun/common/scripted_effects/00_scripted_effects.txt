			give_nickname = nick_the_recluse

			mult_modifier = {
				factor = 2
				trait = shy
			}
		}

		10 = {
			trigger = { trait = wroth }

			give_nickname = nick_the_short_tempered
		}

		10 = {
			trigger = {
				NOT = { trait = content }
			}

			random_list = {
				10 = { give_nickname = nick_the_whiner }
				10 = { give_nickname = nick_the_malcontent }
			}
		}

		10 = {
			trigger = {
				trait = deceitful
			}

			random_list = {
				10 = { give_nickname = nick_the_liar }
				10 = { give_nickname = nick_the_deceitful }
				10 = { give_nickname = nick_the_deceiver }
			}
		}
	}
