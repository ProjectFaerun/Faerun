focus_ch_heritage = {
	type = character

	modifier_name = CHILDHOOD_FOCUS_HERITAGE

	potential = {
		age < 12
		NOR = {
			culture_group = draconic_culture_group
			culture_group = celestial_group 
			culture_group = genie_group
			culture_group = construct_group
		}
		NOT = {
			liege = {
				OR = {
					culture_group = draconic_culture_group
					culture_group = celestial_group 
					culture_group = genie_group
					culture_group = construct_group
				}				
			}
		}
	}

	# Focuses have no success conditions
	success = {
		always = no
	}

	abort = {
		age >= 12
	}

	# AI Pick chance
	chance = {
		factor = 500

		trigger = {
			is_ruler = no

			NOT = {
				liege = {
					culture = ROOT
					religion = ROOT
				}
			}
		}
	}
}
