olrox_all_immortals_trigger = { #both the longlived and the completely immortal
	custom_tooltip = {
		text = olrox_all_immortals_trigger_tooltip

		trigger_if = {
			limit = { age < 2 } # newborn characters can't have immortal_age property, so we use heavy triggers for them to give them time to get it through racial_aging.3 event at the age of 1

			OR = {
				BR_long_lived_races = yes
				faerun_immortal_race_trigger = yes
			}
		}
		trigger_else = {
			immortal = yes
		}
	}
}
olrox_not_immortals_trigger = { #opposite of olrox_all_immortals_trigger, made for tooltip reasons
	custom_tooltip = {
		text = olrox_not_immortals_trigger_tooltip

		trigger_if = {
			limit = { age < 2 } # newborn characters can't have immortal_age property, so we use heavy triggers for them to give them time to get it through racial_aging.3 event at the age of 1

			NOR = {
				BR_long_lived_races = yes
				faerun_immortal_race_trigger = yes
			}
		}
		trigger_else = {
			immortal = no
		}
	}
}

is_secretly_undead = {
	trigger_if = {
		limit = {
			OR = {
				trait = vampire
				trait = vampire_spawn
			}
		}

		NOT = { has_character_modifier = known_vamp_modifier }
	}
	trigger_else_if = {
		limit = { has_character_flag = secret_lich }

		is_undead = no
	}
	trigger_else = {
		always = no
	}
}
