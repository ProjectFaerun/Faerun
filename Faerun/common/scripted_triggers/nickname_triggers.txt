can_apply_lame_nickname_trigger = {
	has_nickname = no
}

can_apply_average_nickname_trigger = {
	trigger_if = {
		limit = { has_nickname = yes }
		has_lame_nickname_trigger = yes
	}
}

can_apply_cool_nickname_trigger = {
	trigger_if = {
		limit = { has_nickname = yes }

		OR = {
			has_lame_nickname_trigger = yes
			has_average_nickname_trigger = yes
		}
	}
}

can_apply_super_cool_nickname_trigger = {
	trigger_if = {
		limit = { has_nickname = yes }

		OR = {
			has_lame_nickname_trigger = yes
			has_average_nickname_trigger = yes
			has_cool_nickname_trigger = yes
		}
	}
}
