holy_site_spread = {
	factor = 100
trigger = {
		# At least one holy site is guaranteed to be within the religion's area
		trigger_if = {
			limit = { FROM = { num_holy_sites_generated < 1 } }
			religion = FROM
		}

		# Avoid clustering
		holy_site_distance = {
			target = FROM
			value >= 100
		}

		# We use this as a proxy for religion distance, because religion distance is expensive to calculate
		OR = {
			religion = FROM

			# Heresies can put their 3rd holy site in their parent religion
			AND = {
				parent_religion = { religion = FROM }
				FROM = { num_holy_sites_generated < 3 }
			}

			FROM = { num_holy_sites_generated < 1 }

			holy_site_distance = {
				target = FROM
				value <= 400
			}
		}

		# We simply never want to be adjacent to a holy site
		NOT = {
			any_neighbor_province = {
				is_holy_site = FROM
			}
		}
	}

	# We want three holy sites within the religion's borders, and two outside
	mult_modifier = {
		factor = 0.01
						   
												 
		FROM = { num_holy_sites_generated < 3 }

		NOR = {
			religion = FROM
			parent_religion = {
				religion = FROM
			}
		}
	}
	mult_modifier = {
		factor = 0.001
		FROM = { num_holy_sites_generated < 3 }
		NOT = { religion_group = FROM }
										 
	}
	mult_modifier = {
		factor = 0.001
		FROM = { num_holy_sites_generated >= 3 }
		religion = FROM
	}

	# Stay near our borders
	mult_modifier = {
		factor = 2
		religion_group = FROM
	}

	# Avoid clustering
	mult_modifier = {
		factor = 0.01
		holy_site_distance = {
			target = FROM
			value < 200
		}
	}
	# mult_modifier = {
		# factor = 0.1
		# holy_site_distance = {
			# target = FROM
			# value < 300
		# }
		# FROM = {
			# NOT = { religion_group = pagan_group }
													
		# }
													 
	# }
	mult_modifier = {
		factor = 20
													
		FROM = { num_holy_sites_generated > 0 }
		holy_site_distance = {
			target = FROM
			value > 500
		}
	}
