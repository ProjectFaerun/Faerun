#Retinue General Balance
#MAIN TROOP TYPE gets 8 "points" SECOND TROOP TYPE gets 6 "points" (exceptions exist, like basic spellcasting retinues getting 10 points, and elite spellcasters 30 points)
#SKIRMISH ORIENTED TROOP: OFF = 0.075*points MOR = 0.1*points DEF = 0.15*points
#MELEE ORIENTED TROOP: MOR = 0.075*points OFF = 0.1*points DEF = 0.15*points

##Generic Giant##
retinuegiant = {
	first_type = 6
	first_amount = 20
	second_type = 1
	second_amount = 80

	special_troops = giant_troops

	potential = {
		is_giant_race = yes
		NOT = { has_character_modifier = undead_realm_lord }
		NOT = { government = nomadic_tribal_government }
		OR = {
			higher_tier_than = BARON
			is_patrician = yes
		}
	}

	hire_cost = 225
	costs_prestige = yes

	maintenance_multiplier = 0.75

	modifier = {
		giant_troops_offensive = 0.1 #giants are strong and get fewer points
		giant_troops_defensive = 0.1
		heavy_infantry_offensive = 0.3 #3 points
		heavy_infantry_morale = 0.375 #5 points
	}
}
