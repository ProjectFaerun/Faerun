cb_install_khan = {
	name = CB_NAME_INSTALLKHAN
	war_name = WAR_NAME_INSTALLKHAN
	sprite = 12
	truce_days = 3650

	is_revolt_cb = no
	can_call_vassals = yes
	attacker_can_call_allies = no
	major_revolt = yes

	can_use_gui = {
		ROOT = {
			show_scope_change = no
			war_cost = yes
		}
	}

	can_use = {
		has_dlc = "Horse Lords"

		# Attacker
		ROOT = {
			government = nomadic_tribal_government
		}
	}

	is_valid = {
		# Attacker
		ROOT = {
			liege_before_war = { character = FROM }
			num_of_count_titles >= 1
		}

		# Defender
		FROM = {
			higher_real_tier_than = ROOT
		}
	}

	on_add = {
		attacker = {
			show_scope_change = no
			declare_war_effect = yes 
		}
	}

	on_success = {
		attacker = {
			show_scope_change = no
			participation_scaled_prestige = 250
			attacker_victory_effect = yes
		}

		defender = {
			show_scope_change = no

			primary_title = {
				show_scope_change = no
				gain_title = {
					target = attacker
					type = faction_demand
				}
			}

			death = {
				death_reason = death_execution
				killer = attacker
			}
		}

		any_attacker = {
			limit = { NOT = { character = attacker } }
			hidden_effect = { participation_scaled_prestige = 250 }
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no

			prestige = -100
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			prestige = -200
			imprison = defender
		}

		defender = {
			show_scope_change = no

			participation_scaled_prestige = 50
			defender_victory_effect = yes

			hidden_effect = {
				if = {
					limit = {
						NOT = {
							has_opinion_modifier = { # Allow free revocation of a title
								name = opinion_traitor
								who = attacker
							}
						}
					}

					opinion = {
						name = opinion_traitor
						who = attacker
					}
				}
			}
		}

		any_defender = {
			limit = { NOT = { character = defender } }
			hidden_effect = { participation_scaled_prestige = 50 }
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	on_defender_leader_death = {
		any_attacker = {
			letter_event = { id = 251 } # Our enemy has died, the war has ended
		}

		end_war = invalid
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}

# CB for non-nomads to seize a duchy off a nomad horde
cb_liberate_nomad_duchy = {
	name = CB_NAME_LIBERATE_NOMAD
	war_name = WAR_NAME_LIBERATE_NOMAD
	sprite = 35
	truce_days = 3650
	infamy_modifier = 0.75
	sort_priority = 810

	is_permanent = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	allowed_to_target_tributaries = no

	can_use_gui = {
		ROOT = {
			show_scope_change = no
			war_cost = yes
		}
	}

	can_use = {
		# Attacker
		ROOT = {
			NOR = {
				government = nomadic_tribal_government
				same_realm = FROM
				mercenary = yes
				is_pacifist_religion = yes
				trait = creature_monster
			}
		}

		# Defender
		FROM = {
			government = nomadic_tribal_government
			NOT = { trait = creature_monster }
		}
	}

	can_use_title = {
		# The attacker needs at least one county in the target duchy, or a border
		any_direct_de_jure_vassal_title = {
			OR = {
				holder_scope = {
					OR = {
						character = ROOT
						is_vassal_or_below_of = ROOT
					}
				}

				location = {
					any_neighbor_province = {
						has_owner = yes

						owner = {
							OR = {
								character = ROOT
								is_vassal_or_below_of = ROOT
							}
						}
					}
				}
			}
		}

		# Can only liberate Duchies that are either outside of the Steppe region, or contains a Castle/City
		any_de_jure_vassal_title = {
			real_tier = COUNT

			holder_scope = {
				OR = {
					character = FROM
					is_vassal_or_below_of = FROM
				}
			}

			location = {
				OR = {
					nomadic_tribal_region_trigger = no
					has_city = yes
					has_castle = yes
				}
			}
		}
	}

	is_valid = {
		# Attacker
		ROOT = {
			NOR = {
				government = nomadic_tribal_government
				same_realm = FROM
				mercenary = yes
				is_pacifist_religion = yes
			}
		}

		# Defender
		FROM = {
			government = nomadic_tribal_government
		}
	}

	on_add = {
		attacker = {
			show_scope_change = no
			declare_war_effect = yes 
		}
	}

	on_success = {
		attacker = {
			show_scope_change = no

			participation_scaled_prestige = 200
			attacker_victory_effect = yes

			if = {
				limit = { uses_decadence = yes }
				participation_scaled_decadence = -10
			}
		}

		any_attacker = {
			limit = { NOT = { character = attacker } }

			hidden_effect = {
				participation_scaled_prestige = 200

				if = {
					limit = { uses_decadence = yes }
					participation_scaled_decadence = -10
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = -200
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = liberate_nomad_cb_succ_tip

			defender = {
				any_realm_character = {
					limit = {
						is_landed = yes
						higher_real_tier_than = baron
						government = nomadic_tribal_government
					}
					set_character_flag = liberate_nomad_usurp
					character_event = { id = nomadtribalmaint.14 days = 1 }
				}
			}
			attacker = {
				vassalize_or_take_under_title = {
					title = PREV
					enemy = defender
					anti_nomad = yes # Vassalize non-nomadic rulers
					type = invasion
				}
				any_realm_character = {
					limit = {
						has_character_flag = liberate_nomad_usurp
					}
					any_demesne_title = {
						limit = { is_tribal_type_title = no }
						usurp_title = {
							type = invasion
							target = ROOT
						}
					}
				}
			}
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no

			prestige = -200
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			transfer_scaled_wealth = {
				to = defender
				value = 4.0
			}

			prestige = -200
		}

		defender = {
			show_scope_change = no

			participation_scaled_prestige = 300
			defender_victory_effect = yes

			if = {
				limit = { uses_decadence = yes }
				participation_scaled_decadence = -10
			}
		}

		any_defender = {
			limit = { NOT = { character = defender } }

			hidden_effect = {
				participation_scaled_prestige = 300

				if = {
					limit = { uses_decadence = yes }
					participation_scaled_decadence = -10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = {
		factor = 1

		trigger = {
			trigger_if = {
				limit = {
					ROOT = {
						is_tribal = no
					}
				}

				any_direct_de_jure_vassal_title = {
					any_direct_de_jure_vassal_title = {
						OR = {
							holding_type = castle
							holding_type = city
						}
					}
				}
			}
			trigger_else = {
				any_direct_de_jure_vassal_title = {
					any_direct_de_jure_vassal_title = {
						OR = {
							holding_type = castle
							holding_type = city
							holding_type = tribal
						}
					}
				}
			}
		}
	}
}

# The CB used by "Minor Clan" revolters in Horse Lords
cb_minor_clan_revolt = {
	name = CB_NAME_MINOR_CLAN_REVOLT
	war_name = WAR_NAME_MINOR_CLAN_REVOLT
	sprite = 16
	truce_days = 365

	hostile_against_others = no
	is_permanent = yes
	check_all_titles = yes # if permanent, setting this to true will check against all of someones titles, including vassal held titles
	press_claim = yes
	attacker_can_call_allies = no
	defender_can_call_allies = no
	can_ask_to_join_war = no
	is_independence = yes

	can_use = {
		# Attacker
		ROOT = {
			primary_title = { temporary = yes } # Revolter or adventurer trigger

			NOT = { is_vassal_or_below_of = FROM }
			war = no
			mercenary = no
		}
	}

	can_use_title = {
		real_tier = COUNT
	}

	is_valid_title = {
		holder_scope = {
			OR = {
				character = FROM
				is_vassal_or_below_of = FROM
			}
		}
	}

	on_add = {
		fire_haruspicy_event_effect = yes
		anger_china_effect = yes
	}

	on_success_title = {
		usurp_title_plus_barony_if_unlanded = {
			target = attacker
			type = revolt
		}

		attacker = {
			show_scope_change = no
			attacker_victory_effect = yes

			hidden_effect = {
				primary_title = {
					set_title_nomad = {
						title = THIS
						status = no
					}
				}
			}

			set_government_type = nomadic_tribal_government
			set_defacto_liege = FROM

			hidden_effect = {
				remove_opinion = {
					name = opinion_evil_tyrant
					who = defender
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = -100
		}
	}

	on_fail = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					duration = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 10

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}
			}
		}
	}

	on_reverse_demand = {
		attacker = {
			show_scope_change = no

			clear_wealth = yes
			imprison = defender

			hidden_effect = {
				add_character_modifier = {
					name = broken_spirit
					duration = -1
				}
			}
		}

		defender = {
			show_scope_change = no

			prestige = 20
			defender_victory_effect = yes

			hidden_effect = {
				opinion = {
					name = opinion_rebel_traitor
					who = attacker
					years = 100
				}

				add_character_modifier = {
					name = crushed_revolt
					years = 10
				}
			}
		}
	}

	on_attacker_leader_death = {
		end_war = invalid
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}
}
